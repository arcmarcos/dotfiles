### This is abbr feature in fish, who creates aliases with asteroids! Read the docs to understand
### To add more abbreviations, just make as the below ones and execute the script 'create-abbr.sh'
### GIT
abbr gs git status
abbr ga git add
abbr gc git commit -m
abbr gck git checkout
abbr gp git pull
abbr gps git push
abbr gcl git clone
abbr gb git branch
abbr gl git log
### APT
abbr up sudo apt update && sudo apt upgrade -y
### NPM
abbr npmi npm install
abbr npmr npm run
abbr npmd npm run dev
abbr npmsv npm run serve
abbr npmh npm run hot
abbr npmw npm run watch
abbr npmp npm run prod
abbr npmt npm run test
abbr npmrm npm rm
### YARN
abbr ya yarn
abbr yaa yarn add
abbr yar yarn run
### MOVE
abbr down cd ~/Downloads
abbr dot cd ~/dotfiles
### SYSTEM
abbr chx chmod +x
abbr chr chmod -R
### MISC
abbr nf neofetch
### DOCKER
abbr dp docker ps
abbr dpa docker ps -a
abbr dc docker compose
abbr dcp docker compose ps
abbr dcpa docker compose ps -a
